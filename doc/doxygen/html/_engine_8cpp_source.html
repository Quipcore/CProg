192bool Engine::isOutOfBounds(Vector2 &objectPosition, Map &map) {
193 if (objectPosition.getX() < 0 || objectPosition.getX() > SCREEN_WIDTH) {
194 return true;
195 }
196 if (objectPosition.getY() < 0 || objectPosition.getY() > SCREEN_HEIGHT) {
197 return true;
198 }
199
200 int mapWidth = map.getWidth();
201 int mapHeight = map.getHeight();
202 int currentXCell = static_cast<int>(objectPosition.getX() * mapWidth / SCREEN_WIDTH);
203 int currentYCell = static_cast<int>(objectPosition.getY() * mapHeight / SCREEN_HEIGHT);
204 Vector2 objectMapPos = {currentXCell, currentYCell};
205 return map.isOutOfBounds(objectMapPos);
206}
